import uvicorn


from core.config import settings





if __name__ == "__main__":
    print("Server is running at " + settings.SERVER_IP + ":" + settings.SERVER_PORT )
    uvicorn.run(
        "main:app",
        host=settings.SERVER_IP,
        port=int(settings.SERVER_PORT),
        log_level="info",
        reload=True
    )
